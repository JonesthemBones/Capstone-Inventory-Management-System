create table public.backup_logs (
  backup_id uuid not null default extensions.uuid_generate_v4 (),
  backup_type character varying(30) not null,
  backup_location character varying(500) not null,
  backup_size_mb numeric(10, 2) null,
  backup_status character varying(20) not null,
  started_at timestamp with time zone null default now(),
  completed_at timestamp with time zone null,
  performed_by uuid null,
  error_message text null,
  retention_expires_at date null,
  constraint backup_logs_pkey primary key (backup_id),
  constraint backup_logs_backup_status_check check (
    (
      (backup_status)::text = any (
        (
          array[
            'in_progress'::character varying,
            'completed'::character varying,
            'failed'::character varying
          ]
        )::text[]
      )
    )
  ),
  constraint backup_logs_backup_type_check check (
    (
      (backup_type)::text = any (
        (
          array[
            'full'::character varying,
            'incremental'::character varying,
            'receipt_images'::character varying,
            'database'::character varying
          ]
        )::text[]
      )
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_backup_logs_type on public.backup_logs using btree (backup_type) TABLESPACE pg_default;

create index IF not exists idx_backup_logs_started_at on public.backup_logs using btree (started_at) TABLESPACE pg_default;
