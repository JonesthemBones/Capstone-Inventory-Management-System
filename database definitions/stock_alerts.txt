create table public.stock_alerts (
  alert_id uuid not null default extensions.uuid_generate_v4 (),
  product_id uuid not null,
  alert_type character varying(30) not null,
  alert_level character varying(20) null default 'warning'::character varying,
  current_quantity integer null,
  threshold_value integer null,
  alert_date timestamp with time zone null default now(),
  is_acknowledged boolean null default false,
  acknowledged_by uuid null,
  acknowledged_at timestamp with time zone null,
  resolution_action character varying(30) null,
  resolution_notes text null,
  resolved_at timestamp with time zone null,
  notes text null,
  constraint stock_alerts_pkey primary key (alert_id),
  constraint stock_alerts_product_id_fkey foreign KEY (product_id) references products (product_id) on delete CASCADE,
  constraint stock_alerts_alert_level_check check (
    (
      (alert_level)::text = any (
        (
          array[
            'info'::character varying,
            'warning'::character varying,
            'critical'::character varying
          ]
        )::text[]
      )
    )
  ),
  constraint stock_alerts_alert_type_check check (
    (
      (alert_type)::text = any (
        (
          array[
            'low_stock'::character varying,
            'out_of_stock'::character varying,
            'overstock'::character varying,
            'approaching_expiry'::character varying,
            'expired'::character varying
          ]
        )::text[]
      )
    )
  ),
  constraint stock_alerts_resolution_action_check check (
    (
      (resolution_action)::text = any (
        (
          array[
            'restocked'::character varying,
            'order_placed'::character varying,
            'false_alarm'::character varying,
            'other'::character varying
          ]
        )::text[]
      )
    )
  )
) TABLESPACE pg_default;

create index IF not exists idx_stock_alerts_type on public.stock_alerts using btree (alert_type) TABLESPACE pg_default;

create index IF not exists idx_stock_alerts_date on public.stock_alerts using btree (alert_date) TABLESPACE pg_default;

create index IF not exists idx_stock_alerts_acknowledged on public.stock_alerts using btree (is_acknowledged) TABLESPACE pg_default
where
  (is_acknowledged = false);

create index IF not exists idx_stock_alerts_product on public.stock_alerts using btree (product_id) TABLESPACE pg_default;

create index IF not exists idx_stock_alerts_unacknowledged on public.stock_alerts using btree (alert_type, is_acknowledged) TABLESPACE pg_default
where
  (is_acknowledged = false);

create index IF not exists idx_stock_alerts_product_resolved on public.stock_alerts using btree (product_id, resolved_at) TABLESPACE pg_default;

create index IF not exists idx_stock_alerts_unresolved on public.stock_alerts using btree (resolved_at) TABLESPACE pg_default
where
  (resolved_at is null);
