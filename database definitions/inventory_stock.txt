create table public.inventory_stock (
  stock_id uuid not null default extensions.uuid_generate_v4 (),
  product_id uuid not null,
  quantity integer not null default 0,
  last_restock_date timestamp with time zone null,
  last_sale_date timestamp with time zone null,
  updated_at timestamp with time zone null default now(),
  constraint inventory_stock_pkey primary key (stock_id),
  constraint inventory_stock_product_id_key unique (product_id),
  constraint inventory_stock_product_id_fkey foreign KEY (product_id) references products (product_id) on delete CASCADE,
  constraint check_non_negative_stock check ((quantity >= 0))
) TABLESPACE pg_default;

create index IF not exists idx_inventory_stock_product on public.inventory_stock using btree (product_id) TABLESPACE pg_default;

create index IF not exists idx_inventory_stock_quantity on public.inventory_stock using btree (quantity) TABLESPACE pg_default;

create trigger audit_inventory_stock
after
update on inventory_stock for EACH row
execute FUNCTION log_audit_trail ();

create trigger update_inventory_stock_updated_at BEFORE
update on inventory_stock for EACH row
execute FUNCTION update_updated_at_column ();
