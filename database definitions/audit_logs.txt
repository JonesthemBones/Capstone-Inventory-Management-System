create table public.audit_logs (
  log_id uuid not null default extensions.uuid_generate_v4 (),
  user_id uuid null,
  action_type character varying(100) not null,
  table_affected character varying(100) null,
  record_id uuid null,
  old_values jsonb null,
  new_values jsonb null,
  ip_address inet null,
  user_agent text null,
  action_timestamp timestamp with time zone null default now(),
  constraint audit_logs_pkey primary key (log_id)
) TABLESPACE pg_default;

create index IF not exists idx_audit_logs_user on public.audit_logs using btree (user_id, action_timestamp) TABLESPACE pg_default;

create index IF not exists idx_audit_logs_action_type on public.audit_logs using btree (action_type) TABLESPACE pg_default;

create index IF not exists idx_audit_logs_timestamp on public.audit_logs using btree (action_timestamp) TABLESPACE pg_default;

create index IF not exists idx_audit_logs_old_values on public.audit_logs using gin (old_values) TABLESPACE pg_default;

create index IF not exists idx_audit_logs_new_values on public.audit_logs using gin (new_values) TABLESPACE pg_default;

create index IF not exists idx_audit_logs_record_id on public.audit_logs using btree (record_id) TABLESPACE pg_default;

create index IF not exists idx_audit_logs_table_record on public.audit_logs using btree (table_affected, record_id) TABLESPACE pg_default;
