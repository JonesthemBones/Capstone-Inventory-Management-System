create table public.products (
  product_id uuid not null default extensions.uuid_generate_v4 (),
  product_code character varying(50) null,
  product_name character varying(200) not null,
  description text null,
  unit_of_measure character varying(20) not null,
  unit_price numeric(10, 2) null default 0.00,
  selling_price numeric(10, 2) null default 0.00,
  reorder_level integer null default 10,
  maximum_stock integer null default 1000,
  is_active boolean null default true,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  constraint products_pkey primary key (product_id),
  constraint products_product_code_key unique (product_code)
) TABLESPACE pg_default;

create index IF not exists idx_products_active on public.products using btree (is_active) TABLESPACE pg_default
where
  (is_active = true);

create index IF not exists idx_products_code on public.products using btree (product_code) TABLESPACE pg_default;

create index IF not exists idx_products_name on public.products using btree (product_name) TABLESPACE pg_default;

create trigger audit_products
after INSERT
or DELETE
or
update on products for EACH row
execute FUNCTION log_audit_trail ();

create trigger update_products_updated_at BEFORE
update on products for EACH row
execute FUNCTION update_updated_at_column ();
